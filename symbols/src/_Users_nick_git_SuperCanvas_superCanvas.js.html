 19 /**
 20  * @namespace 
 21  * @class Construct a new superCanvas Instance
 22  * @param el reference to the canvas element to use for the context
 23  */
 24 var superCanvas = function(el){
 25     var obj, e;
 26     obj = el.getContext("2d");
 27     for(e in superCanvas){
 28         if(superCanvas.hasOwnProperty(e)){
 29             obj[e] = superCanvas[e];
 30         }
 31     }
 32     obj.external = el;
 33     return obj;
 34 };
 35 /**
 36  * @description parses the SVG path for use
 37  * @static
 38  * @param {String}d the path to process
 39  * @returns {Array[]} A 2D array of SVG commands
 40  */
 41 superCanvas.parsePath = function(d){
 42 	d = d.replace(/([mlhvcqtzs])/ig, " $1 ");
 43 	var splitPath = d.match(/([mlhvcqtzs][^mlhvcqtzs]*)/ig),
 44 	pathArr = [], i = 0;
 45 	for(i = 0; i!==splitPath.length; i++){
 46 	    //command = splitPath[i].match(/[\-0-9e]?([^ ,]+)/ig);
 47 	    command = splitPath[i].match(/([\-]?(0|[1-9]\d*)(\.\d*)?([eE][+\-]?\d+)?|[mlhvcqtzs])/ig);
 48 	    //command = splitPath[i].match(/([\-0-9e])?([^ ,\-](e[\-])?)+/ig);
 49 	    //console.log(splitPath[i]);
 50 	    pathArr.push(command);
 51 	}
 52 	return superCanvas.normalizePath(pathArr.slice());
 53 };
 54 /**
 55  * @description just fills text in upside down, really only useful if you are trying to debug SVG font glyphs
 56  * @param text the string to fill
 57  * @param x the x coordinate for the text
 58  * @param y the y coordinate for the text
 59  * @param maxWidth the maximum width for the text
 60  */
 61 superCanvas.fillTextU = function(text, x, y, maxWidth){
 62 this.save();
 63 this.translate(x, y);
 64 this.scale(1,-1);
 65 this.fillText(text, 0,0, maxWidth);
 66 this.restore();
 67 };

179  * @description draws a path created with superCanvas.parsePath
180  * @param dArr the path to draw. Can be either a SVG path or a array created with SuperCanvas.parsePath

184 		dArr = this.parsePath(dArr);
185 	}
186 	this.cX = [];
187 	this.cY = [];
188 	this.currentPath = [];
189 	this.lastCommand = '';
190 	var d = dArr.slice(0),
191     centerX=0, centerY=0, i;
192 	for(i = 0; i!==d.length; i++){
193         var c = d[i].slice(0);
194 		this.currentPath.push(c.slice());
195 		command = c.shift();
196 		//while(this.pathLengths[command.toUpperCase()] <= c.length){
197             //var cee = c.splice(0,this.pathLengths[command.toUpperCase()]);
198             //console.log(cee);
199             try  {
200             var C = this[superCanvas.pathCommands[command]].apply(this, c);
201              this.cX.push(C[0]);
202              this.cY.push(C[1]);
203 	    }catch(e){
204 		console.error(command);
205 		console.error(this[superCanvas.pathCommands[command]]);
206 		throw command;
207 	    }
208 		    this.lastCommand = command;
209             centerX += parseFloat(this.cX[this.cX.length-1]);
210             centerY += parseFloat(this.cY[this.cY.length-1]);
211         //}
212 	}
213     return [centerX/this.cX.length, centerY/this.cY.length];
214 };

225 	pathD.normalized = true;

274                 case 'l':
275                 case 'm':

311 /**
312  * @description resize a Image, Canvas or Video Element without blurring (at time of writng, chrome can't do this with CSS)
313  * @function
314  * @param imageElement the Image, Canvas or Video Element to be scaled
315  * @param w the new width for the element
316  * @param h the new height for the element
317 */
318 superCanvas.resizeImage = function(imageElement, w, h){
319     image=document.createElement("canvas");
320     image.width = imageElement.width;
321     image.height = imageElement.height;
322     ctx = image.getContext("2d");
323     ctx.drawImage(imageElement, 0,0,imageElement.width,imageElement.height);
324     imgData = ctx.getImageData(0,0,image.width,image.height);
325     image.width = w;
326     image.height= h;
327     ctx.scale((w/imageElement.width),(h/imageElement.height));
328     var leewayX = imageElement.width/w;
329     var leewayY = imageElement.height/h;
330     for(var x = 0; x<imgData.width; x++){
331         for(var y = 0; y<imgData.height; y++){
332             var p = (x + y*imgData.width)*4;
333             var P = [].slice.call(imgData.data,p, p+4);
334             ctx.fillStyle="rgba("+P.join(',')+")";
335             ctx.fillRect(x,y,1+leewayX,1+leewayY);
336         }
337     }
338     return image;
339 };
