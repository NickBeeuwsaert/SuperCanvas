 20  * @description wooo!
 21  * @class Construct a new superCanvas Instance
 22  * @constructor
 23  * @param el reference to the canvas element to use for the context
 24  */
 25 var superCanvas = function(el){
 26     var obj, e;
 27     obj = el.getContext("2d");
 28     for(e in superCanvas){
 29         if(superCanvas.hasOwnProperty(e)){
 30             obj[e] = superCanvas[e];
 31         }
 32     }
 33     obj.external = el;
 34     return obj;
 35 };

 42 superCanvas.parsePath = function(d){
 43 	d = d.replace(/([mlhvcqtzsa])/ig, " $1 ");
 44 	var splitPath = d.match(/([mlhvcqtzsa][^mlhvcqtzsa]*)/ig),
 45 	pathArr = [], i = 0;
 46 	for(i = 0; i!==splitPath.length; i++){
 47 	    //command = splitPath[i].match(/[\-0-9e]?([^ ,]+)/ig);
 48 	    command = splitPath[i].match(/([\-]?(0|[1-9]\d*)(\.\d*)?([eE][+\-]?\d+)?|[mlhvcqtzsa]+)/ig);
 49 	    //command = splitPath[i].match(/([\-0-9e])?([^ ,\-](e[\-])?)+/ig);
 50 	    //console.log(splitPath[i]);
 51 	    pathArr.push(command);
 52 	}
 53 	return superCanvas.normalizePath(pathArr.slice());
 54 };

175 superCanvas.eArc = function( rx, ry, theta, fA, fS, x2,y2){
176     if(rx == 0 || ry == 0) {return this.line2(x2,y2);}
177     var x1 = parseFloat(this.cX[this.cX.length-1]);
178     var y1 = parseFloat(this.cY[this.cY.length-1]);
179     rx = parseFloat(rx);
180     ry = parseFloat(ry);
181     theta = parseFloat(theta);
182     fA = parseFloat(fA);
183     fS = parseFloat(fS);
184     x2 = parseFloat(x2);
185     y2 = parseFloat(y2);
186 
187     theta *= (Math.PI/180); // SVG spec says that the angle passed should be in degrees, so convert from that
188     theta %= Math.PI*2;
189     var mx = (x1-x2)/2;
190     var my = (y1-y2)/2;
191     var Mx = (x1+x2)/2;
192     var My = (y1+y2)/2;
193     x1p = Math.cos(theta) * mx + Math.sin(theta)*my;
194     y1p =-Math.sin(theta) * mx + Math.cos(theta)*my;
195 
196     rx = Math.abs(rx);
197     ry = Math.abs(ry);
198     D = Math.pow(x1p,2)/ Math.pow(rx,2) + Math.pow(y1p,2)/Math.pow(ry,2);
199     //if D is greater than one, the radii won't fit the ellipse that starts at the specified points with the specified angle
200     if (D > 1){
201         rx = Math.sqrt(D)*rx;
202         ry = Math.sqrt(D)*ry;
203     }
204     var c = Math.sqrt(Math.abs((Math.pow(rx,2)*Math.pow(ry, 2) - Math.pow(rx,2)*Math.pow(y1p,2) - Math.pow(ry,2)*Math.pow(x1p,2))/
205               (Math.pow(rx,2)*Math.pow(y1p,2) + Math.pow(ry,2)*Math.pow(x1p,2))));
206     
207     if(fS == fA){
208         c= -c;
209     }
210     cxp = c;
211     cyp = c;
212 
213     cxp *= (rx*y1p)/ry;
214     cyp *= -((ry*x1p)/rx);
215     
216     
217     var cx = Math.cos(theta)*cxp - Math.sin(theta)*cyp;
218     var cy = Math.sin(theta)*cxp + Math.cos(theta)*cyp;
219     cx += Mx;
220     cy += My;
221 
222 
223     // begin calculating theta1 and Dtheta...
224     // theta1 is the starting angle,
225     // and dTheta is difference betwix the end and the start
226     function norm(v){
227         var norm = 0;
228         for(var i = 0; i < v.length;i++){
229             norm += Math.pow(Math.abs(v[i]), 2);
230         }
231         return Math.sqrt(norm);
232     }
233     function getAngle(u, v){
234         var sign = u[0] * v[1] - u[1]*v[0] < 0 ? -1: 1;
235         var toBe = u[0]*v[0] + u[1]*v[1];
236         var orNot= norm(u)*norm(v);
237         var r = toBe/orNot;
238         //correct for floating point errors
239         // stuff like 1.000000009 and all that
240         if(r < -1){
241             r = -1;
242         }else if(r > 1){
243             r = 1;
244         }
245         var res = Math.acos(r);
246         return sign * res;
247     }
248     var theta1 = getAngle([1, 0], [(x1p-cxp)/rx,(y1p-cyp)/ry]);
249     var Dtheta = getAngle([(x1p-cxp)/rx,(y1p-cyp)/ry],[(-x1p-cxp)/rx,(-y1p-cyp)/ry] )%(Math.PI*2);
250     if(fS == 0 && Dtheta > 0){
251         Dtheta -= Math.PI*2;
252     }else if(fS == 1 && Dtheta < 0){
253         Dtheta += Math.PI*2;
254     }
255     var theta2 = Dtheta + theta1;
256 
257     var x = x1;
258     var y = y2;
259     for(var i = 0; i < Math.abs(Dtheta); i+=Math.PI/180){
260 
261         var I = theta1+(Dtheta<0?-1:1)*i;
262         x = cx + (Math.cos(I)*rx*Math.cos(theta) - Math.sin(I)*ry*Math.sin(theta));
263         y = cy + (Math.cos(I)*rx*Math.sin(theta) + Math.sin(I)*ry*Math.cos(theta));
264         this.lineTo(x, y);
265     }
266     this.lineTo(x2,y2);
267     return [x2, y2];
268 };

295 /**
296  * @description a matrix class
297  * @class Matrices! Weeee!
298  * @param [a] m11, defaults to 1
299  * @param [b] m21, defaults to 0
300  * @param [c] m12, defaults to 0
301  * @param [d] m22, defaults to 1
302  * @param [e] m13, defaults to 0
303  * @param [f] m23, defaults to 0
304  */
305 superCanvas.Matrix = function(a,b,c,d,e,f){
306     var t = {};
307 
308 
309     /**
310      * @description multiplies the matrix by the given areguments 
311      * @memberOf superCanvas.Matrix
312      * @param a m11
313      * @param b m21
314      * @param c m12
315      * @param d m22
316      * @param e m13
317      * @param f m23
318      * @returns this instance of the superCanvas.Matrix
319      */
320     transform = function(a,b,c,d,e,f){
321         var m = superCanvas.Matrix(a,b,c,d,e,f);
322         var m11, m21, m12, m22, m13,m23;
323         m11 = t.m11 * m.m11 + t.m12 * m.m21 + t.m13 * m.m31;
324         m21 = t.m21 * m.m11 + t.m22 * m.m21 + t.m23 * m.m31;
325         
326         m12 = t.m11 * m.m12 + t.m12 * m.m22 + t.m13 * m.m32;
327         m22 = t.m21 * m.m12 + t.m22 * m.m22 + t.m23 * m.m32;
328         
329         m13 = t.m11 * m.m13 + t.m12 * m.m23 + t.m13 * m.m33;
330         m23 = t.m21 * m.m13 + t.m22 * m.m23 + t.m23 * m.m33;
331         
332         t.m11 = m11;
333         t.m21 = m21;
334         
335         t.m12 = m12;
336         t.m22 = m22;
337         
338         t.m13 = m13;
339         t.m23 = m23;
340         return t;
341         
342     };
343     /**
344      * @description rotates the matrix by theta degrees
345      * @memberOf superCanvas.Matrix
346      * @param theta radians to rotate the matrix
347      * @returns this instance of the superCanvas.Matrix
348      */
349     rotate = function(theta){
350         var a = Math.cos(theta);
351         var b = Math.sin(theta);
352         var c = -Math.sin(theta);
353         var d = Math.cos(theta);
354         var e = 0, f = 0;
355         t.transform(a,b,c,d,e,f);
356         return t;
357     };
358     /**
359      * @description skews the matrix on the X axis
360      * @memberOf superCanvas.Matrix
361      * @param x amount to skew the matrix
362      * @returns this instance of the superCanvas.Matrix
363      */
364     skewX = function(x){
365         t.transform(1,0,Math.tan(x),1,0,0);
366         return t;
367     };
368     /**
369      * @description skews the matrix on the Y axis
370      * @memberOf superCanvas.Matrix
371      * @param y amount to skew
372      * @returns this instance of the superCanvas.Matrix
373      */
374     skewY = function(y){
375         t.transform(1,Math.tan(y),0,1,0,0);
376         return t;
377     };
378     /** 
379      * @description skews in both axis
380      * @memberOf superCanvas.Matrix
381      * @param x amount to skew on the X axis
382      * @param y amount to skew on the Y axis
383      * @returns this instance of the superCanvas.Matrix
384      */
385     skew = function(x, y){
386         t.transform(1,Math.tan(y),Math.tan(x),1,0,0);
387         return t;
388     };
389     /**
390      * @description translates the matrix
391      * @memberOf superCanvas.Matrix
392      * @param x amount to translate in the X axis
393      * @param y amount to translate in the Y axis
394      * @returns this instance of the superCanvas.Matrix
395      */
396     translate = function(x, y){
397         t.transform(1,0,0,1,x,y);
398         //t.m13 += x;
399         //t.m23 += y;
400         return t;
401     };
402     /**
403      * @description sets the transformation matrix to the values specified by a,b,c,d,e and f
404      * @memberOf superCanvas.Matrix
405      * @param a m11
406      * @param b m21
407      * @param c m12
408      * @param d m22
409      * @param e m13
410      * @param f m23
411      * @returns this instance of the superCanvas.Matrix
412      */
413     setTransform = function(a,b,c,d,e,f){
414         t.m11 = a;
415         t.m21 = b;
416         
417 
418         t.m12 = c;
419         t.m22 = d;
420 
421         t.m13 = e;
422         t.m23 = f;
423 
424         t.m31 = 0;
425         t.m32 = 0;
426         t.m33 = 1;
427         return t;
428     };
429     /**
430      * @description creates a string representation of the matrix
431      * @memberOf superCanvas.Matrix
432      */
433     toString = function(){
434         return [[t.m11, t.m12, t.m13],
435                 [t.m21, t.m22, t.m23],
436                 [t.m31, t.m32, t.m33]].join('\n');
437     };
438     /**
439      * @description sets the transformation matrix to 1,0,0,1,0,0
440      * @memberOf superCanvas.Matrix
441      * @returns this instance of the superCanvas.Matrix
442      */
443     resetTransform = function(){
444         t.setTransform(1,0,0,1,0,0);
445         return t;
446     };
447     /**
448      * @description scales the matrix
449      * @memberOf superCanvas.Matrix
450      * @param x amount to scale the X axis
451      * @param y amount to scale the Y axis
452      * @returns this instance of the superCanvas.Matrix
453      */
454     scale = function(x, y){
455         //t.transform(x,0,0,y,0,0);
456         t.m11 *= x;
457         t.m22 *= y;
458         return t;
459     };
460     /**
461      * @memberOf superCanvas.Matrix
462      * @description multiplies the matrix by the coordinates provided
463      * @param x the x coordinate
464      * @param y the y coordinate
465      */
466     bake = function(x, y){
467         // [ a c e ]   [x]
468         // [ b d f ] * [y]
469         // [ 0 0 1 ]   [1]
470         var X = x * t.m11 + y * t.m12 + 1 * t.m13;
471         var Y = x * t.m21 + y * t.m22 + 1 * t.m23;
472         return [X,Y];
473     };
474     t.rotate = rotate;
475     t.bake = bake;
476     t.scale = scale;
477     t.resetTransform = resetTransform;
478     t.toString = toString;
479     t.setTransform = setTransform;
480     t.translate = translate;
481     t.skew = skew;
482     t.skewX = skewX;
483     t.skewY = skewY;
484     t.transform = transform
485     t.setTransform(a||1,b||0,c||0,d||1,e||0,f||0);
486     return t;
487 };

498     }

502             // I *think* that some transform should be applied the the arcs x and y radius, but I don't know how to do that
503             coords = matrix.bake(path[i][path[i].length-2], path[i][path[i].length-1]);
504             cmd = cmd.concat(path[i].slice(1,-2), coords);
505 
506         }else{
507             for(var I = 0; I<path[i].length/2-1;I++){
508                 cmd = cmd.concat(matrix.bake(path[i][I*2+1], path[i][I*2+2]));
509             }
510         }
511         newPath.push(cmd);
512     }
513     //console.log(newPath);
514     return newPath;
515 };

754 /**
755  * @description resize a Image, Canvas or Video Element without blurring (at time of writng, chrome can't do this with CSS)
756  * @function
757  * @param imageElement the Image, Canvas or Video Element to be scaled
758  * @param w the new width for the element
759  * @param h the new height for the element
760 */
761 superCanvas.resizeImage = function(imageElement, w, h){
762     image=document.createElement("canvas");
763     image.width = imageElement.width;
764     image.height = imageElement.height;
765     ctx = image.getContext("2d");
766     ctx.drawImage(imageElement, 0,0,imageElement.width,imageElement.height);
767     imgData = ctx.getImageData(0,0,image.width,image.height);
768     image.width = w;
769     image.height= h;
770     ctx.scale((w/imageElement.width),(h/imageElement.height));
771     var leewayX = imageElement.width/w;
772     var leewayY = imageElement.height/h;
773     for(var x = 0; x<imgData.width; x++){
774         for(var y = 0; y<imgData.height; y++){
775             var p = (x + y*imgData.width)*4;
776             var P = [].slice.call(imgData.data,p, p+4);
777             ctx.fillStyle="rgba("+P.join(',')+")";
778             ctx.fillRect(x,y,1+leewayX,1+leewayY);
779         }
780     }
781     return image;
782 };
